% ============= ABSTRACT =============
\begin{abstract}
Báo cáo này trình bày nghiên cứu và cài đặt thuật toán \textbf{LCMFreq} cho bài toán \gls{fim}.
LCMFreq, được đề xuất bởi Uno, Kiyomi và Arimura (2004), sử dụng \gls{tidset} theo định dạng dọc (vertical data format) và tìm kiếm theo chiều sâu (\gls{dfs}) để liệt kê toàn bộ các \gls{itemset} phổ biến mà không sinh ứng viên (candidate) tường minh.

Cài đặt của nhóm được viết bằng ngôn ngữ Julia 1.12 với tối ưu hóa then chốt: thay thế TID-set dạng danh sách số nguyên bằng \gls{bitarray}, cho phép tính giao bằng phép AND theo bit (\gls{simd}) và đếm \gls{support} bằng lệnh \gls{popcnt} phần cứng. Kết quả thực nghiệm trên 9 tập dữ liệu chuẩn (Chess, Connect, Mushroom, Pumsb, Accidents, Retail, T10I4D100K, T40I10D100K, Kosarak) cho thấy kết quả khớp 100\% với \gls{spmf} và tốc độ vượt trội từ $3{,}5\times$ đến $141\times$ so với SPMF Java, đồng thời tiết kiệm bộ nhớ tới $27\times$ so với cài đặt cơ sở (baseline).

\textbf{Từ khóa:} Frequent Itemset Mining, LCMFreq, TID-set, BitArray, Julia, SPMF, Khai thác dữ liệu.
\end{abstract}

\section{NỀN TẢNG LÝ THUYẾT}
% ============================================================

\subsection{Bài toán Frequent Itemset Mining}

\subsubsection{Các định nghĩa hình thức}

\begin{definition}[Transaction Database]
Cho $\mathcal{I} = \{i_1, i_2, \ldots, i_m\}$ là tập hữu hạn các \textbf{item}. Một \gls{txdb} là tập $\mathcal{D} = \{T_1, T_2, \ldots, T_n\}$ trong đó mỗi transaction $T_k \subseteq \mathcal{I}$, được gán định danh duy nhất $\mathrm{tid}(T_k) = k$.
\end{definition}

\begin{definition}[Support]
Cho \gls{itemset} $X \subseteq \mathcal{I}$. \textbf{Support tuyệt đối} của $X$ là:
\[
    \supp(X) = \bigl|\{T \in \mathcal{D} : X \subseteq T\}\bigr|
\]
\textbf{Support tương đối} (relative support) là:
\[
    \supp_{\mathrm{rel}}(X) = \frac{\supp(X)}{|\mathcal{D}|}
\]
\end{definition}

\begin{definition}[Frequent Itemset]
Cho ngưỡng \gls{minsup} $\sigma \in \mathbb{Z}^{+}$. Itemset $X$ là \textbf{frequent} nếu:
\[
    \supp(X) \geq \sigma
\]
Bài toán \gls{fim} yêu cầu tìm tất cả frequent itemset: $\mathcal{F} = \{X \subseteq \mathcal{I} : \supp(X) \geq \sigma\}$.
\end{definition}

\begin{definition}[Closed và Maximal Itemset]
Một frequent itemset $X$ được gọi là \textbf{\gls{closed}} nếu không tồn tại $Y \supset X$ sao cho $\supp(Y) = \supp(X)$. $X$ được gọi là \textbf{\gls{maximal}} nếu không tồn tại $Y \supset X$ là frequent.

Quan hệ bao hàm: $\mathcal{F}_{\max} \subseteq \mathcal{F}_{\mathrm{closed}} \subseteq \mathcal{F}$, trong đó $\mathcal{F}_{\max}$ là tập các maximal itemset và $\mathcal{F}_{\mathrm{closed}}$ là tập các closed itemset.
\end{definition}

\begin{definition}[TID-set]
\textbf{TID-set} (hay tidset) của itemset $X$ là:
\[
    \mathcal{T}(X) = \{k \in \{1,\ldots,n\} : X \subseteq T_k\}
\]
Khi đó $\supp(X) = |\mathcal{T}(X)|$. Với hai itemset $P$ và $\{e\}$ rời nhau, $\mathcal{T}(P \cup \{e\}) = \mathcal{T}(P) \cap \mathcal{T}(\{e\})$.
\end{definition}

\subsubsection{Anti-Monotone Property (Downward Closure)}

\begin{property}[Anti-monotone property]
\label{prop:antimonotone}
Với mọi $Y \subseteq X \subseteq \mathcal{I}$:
\[
    \supp(X) \leq \supp(Y)
\]
\end{property}

\begin{proof}
Theo định nghĩa, $\mathcal{T}(X) = \{T \in \mathcal{D} : X \subseteq T\}$ và $\mathcal{T}(Y) = \{T \in \mathcal{D} : Y \subseteq T\}$. Vì $Y \subseteq X$, nên $X \subseteq T$ kéo theo $Y \subseteq T$, tức là $\mathcal{T}(X) \subseteq \mathcal{T}(Y)$. Do đó $\supp(X) = |\mathcal{T}(X)| \leq |\mathcal{T}(Y)| = \supp(Y)$.
\end{proof}

\textbf{Hệ quả (cắt tỉa Apriori).} Nếu $\supp(X) < \sigma$ thì với mọi $Z \supset X$, $\supp(Z) \leq \supp(X) < \sigma$, nghĩa là $Z$ không thể là frequent. Anti-monotone property cho phép loại bỏ toàn bộ cây con candidate của $X$ mà không cần kiểm tra.

\subsection{Phân tích Thuật toán LCMFreq theo Bài báo Gốc}

\subsubsection{Bối cảnh Lịch sử và Hạn chế các Thuật toán Trước}

\gls{lcmfreq} được đề xuất bởi Uno, Kiyomi và Arimura trong bài báo \textit{``LCM ver.2: Efficient Mining Algorithms for Frequent/Closed/Maximal Itemsets''} (FIMI 2004)~\cite{uno2004}. Tên LCM là viết tắt của \textit{Linear time Closed itemset Miner}; LCMfreq là biến thể liệt kê \textbf{tất cả} frequent itemset (không chỉ closed itemset).

Các thuật toán tiền nhiệm thuộc hai nhóm chính:

\begin{itemize}
    \item \textbf{Nhóm Apriori (breadth-first)}: \gls{apriori} (1994) và các biến thể sinh candidate theo từng cấp $k$, quét toàn bộ database để đếm support ở mỗi cấp. Chi phí quét đa lượt và lưu trữ candidate tăng nhanh khi minsup thấp hoặc dữ liệu dense. \gls{fpgrowth} (2000) dùng FP-tree để nén database, tránh quét lại, nhưng cây tốn bộ nhớ và khó cập nhật động.

    \item \textbf{Nhóm Backtracking (DFS)}: \gls{eclat} (1997) dùng định dạng dọc (vertical) -- lưu tidset $\mathcal{T}(e)$ cho từng item $e$. Support của itemset tính bằng giao các tidset, không quét lại database. Hạn chế: với dense dataset, mỗi tidset có thể lên tới $n$ phần tử, tốn nhiều bộ nhớ.
\end{itemize}

Theo Uno et al.~\cite{uno2004}: \textit{``Backtracking algorithms use less memory since they keep only the current solution. Backtracking algorithms need no computation for maintaining previously obtained itemsets, so the computation time of backtracking algorithms is generally short.''} LCMfreq chọn khung backtracking và phát triển thêm hai kỹ thuật then chốt: \\
\begin{center}
    ``\textbf{OccurrenceDeliver} và \textbf{HypercubeDecomposition}.''
\end{center}

\subsubsection{Khung Thuật toán BackTracking}

Theo bài báo gốc (Uno et al., 2004, Section 3.1), khung thuật toán backtracking cho FIM như sau: mỗi lần gọi đệ quy nhận một frequent itemset $P$ (nghiệm hiện tại) và thêm lần lượt từng item $e > \mathrm{tail}(P)$ vào $P$.

\begin{algorithm}[H]
\caption{BackTracking -- Khung cơ bản (Uno et al., 2004, Section 3.1)}
\label{alg:backtracking}
\begin{algorithmic}[1]
\Require Frequent itemset hiện tại $P$
\State Output $P$
\For{mỗi item $e \in \mathcal{I}$, $e > \mathrm{tail}(P)$}
    \If{$P \cup \{e\}$ là frequent}
        \State \Call{BackTracking}{$P \cup \{e\}$}
    \EndIf
\EndFor
\end{algorithmic}
\end{algorithm}

Trong đó $\mathrm{tail}(P)$ là item có chỉ số lớn nhất trong $P$. Quy tắc ``chỉ thêm item lớn hơn tail'' đảm bảo mỗi frequent itemset được liệt kê đúng một lần, không trùng lặp. Khung này dùng \textbf{ít bộ nhớ hơn Apriori} vì chỉ giữ nghiệm hiện tại trên stack, không cần lưu toàn bộ candidate.

\subsubsection{OccurrenceDeliver, Kỹ thuật Đếm Tần suất Hiệu quả}

Bước tốn kém nhất của bất kỳ thuật toán FIM nào là \textbf{đếm support} (frequency counting). Các thuật toán Apriori dùng \textit{down project}: $\mathcal{T}(P) = \mathcal{T}(P_1) \cap \mathcal{T}(P_2)$ (tính giao hai tidset con). Chi phí là $O(|\mathcal{T}(P_1)| + |\mathcal{T}(P_2)|)$ cho mỗi candidate riêng lẻ.

LCMfreq dùng kỹ thuật \textbf{OccurrenceDeliver} (Uno et al., 2004, Section 3.4): thay vì tính giao cho từng candidate $e$ riêng lẻ, thuật toán duyệt qua \textbf{một lần} toàn bộ các transaction trong $\mathcal{T}(P)$ và phân phối chúng vào các \textit{bucket} tương ứng với từng item $e > \mathrm{tail}(P)$.

\begin{algorithm}[H]
\caption{OccurrenceDeliver: Đếm support tất cả candidate cùng lúc (Uno et al., 2004, Section 3.4)}
\label{alg:occdeliver}
\begin{algorithmic}[1]
\Require Database $\mathcal{D}$, frequent itemset hiện tại $P$
\Ensure $\mathrm{Bucket}[e] = \mathcal{T}(P \cup \{e\})$ cho mọi item $e > \mathrm{tail}(P)$
\State Khởi tạo $\mathrm{Bucket}[e] \gets \emptyset$ cho mọi item $e > \mathrm{tail}(P)$
\For{mỗi transaction $t \in \mathcal{T}(P)$}
    \For{mỗi item $e \in t$ với $e > \mathrm{tail}(P)$}
        \State Thêm $t$ vào $\mathrm{Bucket}[e]$
    \EndFor
\EndFor
\State \Return $\{\mathrm{Bucket}[e] : e > \mathrm{tail}(P)\}$
\end{algorithmic}
\end{algorithm}

\textbf{Tại sao hiệu quả?} Sau khi chạy OccurrenceDeliver, $\mathrm{Bucket}[e] = \mathcal{T}(P \cup \{e\})$ cho mọi candidate $e$ cùng lúc, chỉ với một lần duyệt qua $\mathcal{T}(P)$. Chi phí là:
\[
    O\!\Bigl(\sum_{t \in \mathcal{T}(P)} |t \cap \{e : e > \mathrm{tail}(P)\}|\Bigr)
    = O\!\Bigl(|\mathcal{T}(P)| + \sum_{e > \mathrm{tail}(P)} |\mathcal{T}(P \cup \{e\})|\Bigr)
\]
Phức tạp này nhỏ hơn down project vì tránh được chi phí lặp lại $|\mathcal{T}(P)|$ cho mỗi ứng viên.

\subsubsection{HypercubeDecomposition, Xuất Hàng loạt Không cần Đếm Lại}

LCMfreq thêm kỹ thuật \textbf{HypercubeDecomposition} (Uno et al., 2004, Section 3.3) để output nhiều frequent itemset cùng lúc mà không tốn thêm chi phí đếm support:

\begin{definition}[Tập H(P)]
Cho frequent itemset $P$. Định nghĩa:
\[
    H(P) = \{e : e > \mathrm{tail}(P),\ \mathcal{T}(P) = \mathcal{T}(P \cup \{e\})\}
\]
tức là tập tất cả item $e$ lớn hơn tail($P$) sao cho việc thêm $e$ vào $P$ \textbf{không làm giảm support}.
\end{definition}

Với mọi $Q \subseteq H(P)$, ta có $\mathcal{T}(P \cup Q) = \mathcal{T}(P)$, do đó $P \cup Q$ là frequent. Thay vì sinh đệ quy cho mỗi phần tử trong $H(P)$, LCMfreq output toàn bộ các tập con của $P \cup H(P)$ (một \textit{hypercube} trong không gian itemset) chỉ với một lần kiểm tra.

\begin{algorithm}[H]
\caption{HypercubeDecomposition (Uno et al., 2004, Section 3.3)}
\label{alg:hypercube}
\begin{algorithmic}[1]
\Require Frequent itemset hiện tại $P$, tập tích lũy $S$
\State $S' \gets S \cup H(P)$
\State Output tất cả itemset $P'$ thỏa $P \subseteq P' \subseteq P \cup S'$
\Comment{$2^{|S'|}$ itemset, tất cả có support bằng $\mathrm{sup}(P)$}
\For{mỗi item $e \in \mathcal{I} \setminus (P \cup S')$, $e > \mathrm{tail}(P)$}
    \If{$P \cup \{e\}$ là frequent}
        \State \Call{HypercubeDecomposition}{$P \cup \{e\}$, $S'$}
    \EndIf
\EndFor
\end{algorithmic}
\end{algorithm}

Kỹ thuật này tiết kiệm $\approx 2^{|H(P)|}$ lần đếm support -- đặc biệt hiệu quả trên dense dataset nơi nhiều item có cùng tidset.

\subsubsection{Rút gọn CSDL Linh hoạt (Anytime Database Reduction)}

Ngoài hai kỹ thuật trên, LCMfreq dùng \textbf{anytime database reduction} (Uno et al., 2004, Section 3.2): trước mỗi lần gọi đệ quy với $P \cup \{e\}$, thuật toán tạo database rút gọn bằng cách:
\begin{enumerate}
    \item Loại bỏ các transaction \textbf{không chứa} $P \cup \{e\}$ (chúng không bao giờ đóng góp vào kết quả con của $P \cup \{e\}$).
    \item Loại bỏ các item có chỉ số $\leq \mathrm{tail}(P \cup \{e\})$ (chúng không bao giờ được thêm vào nữa).
    \item Hợp nhất các transaction giống hệt nhau.
\end{enumerate}

Database rút gọn này được truyền xuống lần đệ quy tiếp theo, không cần lưu lại database gốc. Kết quả thực nghiệm trong bài báo (Uno et al., 2004, Table 2) cho thấy kỹ thuật này giảm tổng kích thước database tích lũy xuống tới $\mathbf{349\times}$ trên tập Connect và $\mathbf{273\times}$ trên tập Pumsb.

\subsubsection{Cấu trúc Dữ liệu Gốc trong Bài báo}

Điểm quan trọng cần nhấn mạnh: thuật toán LCMfreq nguyên gốc trong bài báo \textbf{không dùng FP-tree} và \textbf{không dùng bitmap/BitArray}. Uno et al.~\cite{uno2004} giải thích:

\begin{quote}
\textit{``We do not use [FP-tree] in the implementation of LCM, but use simple arrays. The main reason is that LCM does not have to search transactions in the database. The main operation of LCM is tracing the transactions in the denotation of the current solution.''}
\end{quote}

Về bitmap, bài báo nhận định: \textit{``The bitmap method has a disadvantage for sparse datasets, and is not orthogonal to anytime database reduction... bitmap method seems to be not good for sparse large datasets.''} Do đó, LCMfreq gốc dùng \textbf{sorted integer arrays} cho tidset và OccurrenceDeliver để đếm support.

\subsubsection{Phân tích Độ phức tạp}

\textbf{Độ phức tạp thời gian.}
\begin{itemize}
    \item \textit{Khởi tạo}: Quét một lượt database để xây dựng bucket ban đầu: $O(\|\mathcal{D}\|)$, trong đó $\|\mathcal{D}\| = \sum_{T \in \mathcal{D}} |T|$ là tổng kích thước database.
    \item \textit{Mỗi nút DFS}: OccurrenceDeliver tốn $O\bigl(\sum_{t \in \mathcal{T}(P)} |t|\bigr)$ cho tất cả candidate cùng lúc, tốt hơn down project $O\bigl(|\mathcal{F}(P)| \cdot |\mathcal{T}(P)|\bigr)$.
    \item \textit{Tổng thể}: Bài báo chứng minh độ phức tạp bị chặn tuyến tính theo số frequent itemset đầu ra $|\mathcal{F}|$: $O(|\mathcal{F}| \cdot \|\mathcal{D}\|)$ trong trường hợp xấu nhất, cải thiện đáng kể so với Apriori.
\end{itemize}

\textbf{Độ phức tạp không gian.}
\begin{itemize}
    \item Lưu trữ database hiện tại: $O(\|\mathcal{D}\|)$ -- không phụ thuộc vào số frequent itemset đã tìm.
    \item Stack đệ quy: mỗi tầng giữ database rút gọn, tổng cộng bị chặn bởi $O(\|\mathcal{D}\| \cdot \mathrm{depth})$.
    \item \textbf{Ưu điểm then chốt}: bộ nhớ không tăng theo $|\mathcal{F}|$, trong khi các thuật toán dạng Apriori phải lưu toàn bộ $D_k$ trong bộ nhớ.
\end{itemize}

\subsection{Cài đặt Baseline, OccurrenceDeliver với Vector tidset}

File \texttt{src/algorithm/lcmfreq\_base.jl} cài đặt phiên bản cơ sở trung thành với tinh thần bài báo gốc: dùng \textbf{OccurrenceDeliver} (bucket-based) để tính support và cấu trúc dữ liệu đơn giản là \texttt{Vector\{Int\}} (danh sách TID sắp xếp tăng dần).

Khởi tạo: xây dựng bucket ban đầu bằng một lần quét toàn bộ database:

\begin{lstlisting}[language=Python, caption={Xây dựng bucket ban đầu bằng OccurrenceDeliver (Baseline)}, label=lst:baseline_bucket]
# Occurrence delivery: buckets[item] = danh sach TID chua item
buckets = [Int[] for _ in 1:(max_item + 1)]
for (tid, tx) in enumerate(transactions)
    for item in tx
        push!(buckets[item], tid)  # phan phoi transaction vao bucket
    end
end
\end{lstlisting}

Tại mỗi bước đệ quy, để tính support của $P \cup \{e\}$ từ bucket hiện tại của $P$, ta lấy giao hai danh sách đã sắp:

\begin{lstlisting}[language=Python, caption={Giao hai TID-list sắp xếp để tính T(P cup {e}) (Baseline)}, label=lst:baseline_intersect]
function intersect_sorted(a::Vector{Int}, b::Vector{Int})::Vector{Int}
    result = Int[]
    i, j = 1, 1
    while i <= length(a) && j <= length(b)
        if a[i] == b[j]
            push!(result, a[i]); i += 1; j += 1
        elseif a[i] < b[j]
            i += 1
        else
            j += 1
        end
    end
    return result
end
\end{lstlisting}

Chi phí giao: $O(|\mathcal{T}(P)| + |\mathcal{T}(e)|)$ mỗi lần. Với $|\mathcal{T}(P)|$ có thể lên tới $n$, chi phí tuyến tính theo $n$ cho mỗi candidate $e$.

\textbf{Hạn chế của Baseline.}
\begin{itemize}
    \item Mỗi phần tử trong TID-list chiếm 8 byte (Int64), trong khi bản chất chỉ cần 1 bit (transaction có/không chứa item).
    \item \texttt{Vector\{Int\}(undef, max\_item+1)} phân bổ mảng theo kích thước ID item lớn nhất -- với sparse dataset như Retail ($\max(\mathcal{I}) = 16{,}470$), gây OOM nghiêm trọng dù chỉ có vài nghìn item phổ biến.
\end{itemize}

\subsection{Cài đặt Tối ưu hóa BitArray}

\textbf{Lưu ý quan trọng:} Tối ưu hóa BitArray \textbf{không có trong bài báo gốc} Uno et al.~\cite{uno2004}. Thực tế, bài báo chỉ rõ rằng bitmap \textit{``has a disadvantage for sparse datasets''} và không khuyến nghị dùng. Đây là đóng góp tối ưu hóa độc lập của nhóm, áp dụng heuristic BitArray trên nền tảng khung backtracking DFS của LCMfreq, phù hợp với môi trường Julia 1.12 có SIMD và POPCNT phần cứng.

File \texttt{src/algorithm/lcmfreq.jl} thay thế TID-list bằng \gls{bitarray} $n$-bit. Các lựa chọn thiết kế chính:

\textbf{(1) Dict thay vì Vector để lưu tidset.} Với dataset thưa, $\max(\mathcal{I})$ có thể rất lớn trong khi số mục phổ biến ít. Dùng \texttt{Dict\{Int, BitArray\{1\}\}} chỉ phân bổ bộ nhớ cho các mục thực sự phổ biến, khắc phục OOM của baseline:

\begin{lstlisting}[caption={Xây dựng tidset BitArray chỉ cho mục phổ biến (Tối ưu)}, label=lst:build_tidset]
tidsets_full = Dict{Int, BitArray{1}}()
for item in all_freq_items
    tidsets_full[item] = falses(n)
end
for (tid, tx) in enumerate(transactions)
    for item in tx
        if haskey(tidsets_full, item)
            @inbounds tidsets_full[item][tid] = true
        end
    end
end
\end{lstlisting}

\textbf{(2) Phép giao bằng AND bit.} Thay vì OccurrenceDeliver, phiên bản tối ưu dùng giao trực tiếp qua phép AND: Julia tự động SIMD-hóa \texttt{.\&} trên BitArray, xử lý 64 transaction mỗi lệnh CPU:

\begin{lstlisting}[caption={Giao tidset bằng AND bit và đếm support bằng POPCNT (Tối ưu)}, label=lst:bitarray_intersect]
tidset_Pe = tidset_P .& tidsets_full[e]   # AND toan bit, tu dong SIMD
support_Pe = count(tidset_Pe)              # POPCNT phan cung
\end{lstlisting}

\subsection{Khả năng Mở rộng (Scalability)}
\textbf{Phương pháp lấy mẫu.}
Thực nghiệm dùng \textbf{uniform random sampling without replacement}
(Fisher-Yates shuffle, seed = 42) để tạo các tập con 10\%--100\% từ Retail.
Đây là phương pháp chuẩn trong benchmark FIM~\cite{goethals2004,fournier2016}
vì transaction database thỏa hai điều kiện:
\begin{enumerate}
    \item \textit{I.I.D.}: mỗi giao dịch là giỏ hàng độc lập, không có
          temporal ordering cần bảo toàn (khác với time-series).
    \item \textit{Probability preservation}:
          $\mathbb{E}[\mathrm{sup}(X,\,\text{sample}_k)]
          = \dfrac{k}{N}\cdot\mathrm{sup}(X,\,\mathcal{D})$
          . Tần suất đồng xuất hiện được bảo toàn theo kỳ vọng, nên
          $\sigma_k = \lceil\text{minsup\_frac}\times k\rceil$ giữ nguyên
          độ khó bài toán qua các mức kích thước.
\end{enumerate}

% Các phương pháp sampling chuyên biệt cho FIM (ISbFIM, ApproxFIM) phục vụ
% mục tiêu khác -- khai thác \textit{approximate} frequent itemset trên big data
% khi không đủ RAM -- không phải benchmark scalability~\cite{goethals2004}.

\begin{enumerate}
    \item \textit{Bộ nhớ cao với T10I4D100K (1.2 GiB):} Vì $n = 100{,}000$, mỗi BitArray chiếm $100{,}000/8 = 12{,}500$ byte. Với nhiều mục phổ biến, tổng bộ nhớ tidset lớn. \end{enumerate}

\subsection{Mô tả Ứng dụng}

\gls{mba} là ứng dụng FIM kinh điển nhất: phân tích hành vi mua sắm của khách hàng để tìm các sản phẩm thường được mua cùng nhau. Kết quả là các \gls{assoc} dạng $X \Rightarrow Y$ với support và \gls{confidence} cao. Ứng dụng được thực hiện trên dataset Retail -- một transaction database thực từ siêu thị Bỉ.

Từ tập frequent itemset $\mathcal{F}$ (khai thác bằng LCMFreq), ta sinh association rule: với mỗi $X \in \mathcal{F}$ và $\emptyset \neq Y \subset X$, luật $Y \Rightarrow (X \setminus Y)$ thỏa mãn nếu:
\begin{align}
    \conf(Y \Rightarrow X \setminus Y) &= \frac{\supp(X)}{\supp(Y)} \geq \text{minconf} \\
    \mathrm{lift}(Y \Rightarrow X \setminus Y) &= \frac{\conf(Y \Rightarrow X \setminus Y)}{\supp_{\mathrm{rel}}(X \setminus Y)} > 1
\end{align}

Với lift $> 1$: các mục trong $Y$ và $X \setminus Y$ có xu hướng đồng xuất hiện nhiều hơn ngẫu nhiên.

\textbf{Nhóm sản phẩm \{38, 39, 41\}.} Lift $> 7$ cho thấy xác suất đồng xuất hiện cao gấp 7 lần so với ngẫu nhiên. \textbf{Nhóm \{32, 48\}.} Lift $\approx 5{,}2$--$6{,}4$ cho thấy mối tương quan đáng kể. Rule $\{32\} \Rightarrow \{48\}$ với confidence 78\%: trong số khách hàng mua item 32, có 78\% cũng mua item 48 -- thông tin hữu ích cho đề xuất sản phẩm (recommendation) thời điểm thanh toán.

\textbf{Tính phổ biến thấp (0.5\%--1.5\%).} Đây là đặc trưng của dataset Retail: rất thưa, hầu hết khách hàng chỉ mua một vài sản phẩm bất kỳ. Tuy nhiên, với 88\,162 transaction, 1\% tương đương 882 khách hàng -- vẫn đủ có ý nghĩa thống kê để đưa ra quyết định kinh doanh.

\textbf{Ứng dụng thực tiễn.}
\begin{enumerate}
    \item \textit{Bố trí kệ hàng:} Đặt sản phẩm 38, 39, 41 gần nhau để tiện lợi và tăng bán chéo.
    \item \textit{Khuyến mãi kết hợp:} Giảm giá gói \{38, 39, 41\} để tăng giá trị đơn hàng.
    \item \textit{Gợi ý sản phẩm online:} Khi khách thêm sản phẩm 32 vào giỏ hàng, hệ thống gợi ý thêm sản phẩm 48.
    \item \textit{Quản lý tồn kho:} Dự báo nhu cầu đồng thời của các sản phẩm liên kết.
\end{enumerate}
