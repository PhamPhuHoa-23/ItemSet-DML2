\begin{frame}{Bài toán Frequent Itemset Mining (FIM)}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{block}{Định nghĩa cơ bản}
        \begin{itemize}\setlength\itemsep{2pt}
          \item $\mathcal{I}$: tập items; $\mathcal{D}$: transaction database
          \item \textbf{Support}: $\mathrm{sup}(X)=|\{T\in\mathcal{D}:X\subseteq T\}|$
          \item $X$ \textbf{frequent} nếu $\mathrm{sup}(X) \geq \sigma$
          \item \textbf{TID-set}: $\mathcal{T}(X)=\{k:X\subseteq T_k\}$
          \item Tính chất: $\mathcal{T}(P\cup\{e\})=\mathcal{T}(P)\cap\mathcal{T}(\{e\})$
        \end{itemize}
      \end{block}
      \vspace{4pt}
      \begin{block}{Dòng thời gian thuật toán FIM}
        \footnotesize
        \begin{tabular}{@{}lll@{}}
          \toprule
          Năm & Thuật toán & Ý tưởng chính \\
          \midrule
          1994 & Apriori & Level-wise candidate gen \\
          1997 & ECLAT   & Vertical tidset format \\
          2000 & FP-Growth & FP-tree, không sinh candidate \\
          \textbf{2004} & \textcolor{psdearteal}{\bfseries LCMFreq} & Backtrack + OccDeliver + Hypercube \\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}

    \begin{column}{0.50\textwidth}
      \begin{alertblock}{Hạn chế của các thuật toán trước}
        \begin{itemize}\setlength\itemsep{2pt}
          \item \textbf{Apriori}: quét DB đa lượt, sinh nhiều candidate tường minh
          \item \textbf{ECLAT}: bộ nhớ lớn với dense dataset
          \item \textbf{FP-Growth}: FP-tree tốn bộ nhớ, khó cập nhật động
        \end{itemize}
      \end{alertblock}
      \vspace{4pt}
      \begin{exampleblock}{LCMFreq giải quyết bằng 3 kỹ thuật}
        \begin{enumerate}\setlength\itemsep{2pt}
          \item \textbf{BackTracking} -- DFS trên prefix tree, ít bộ nhớ
          \item \textbf{OccurrenceDeliver} -- đếm support một lần duyệt
          \item \textbf{HypercubeDecomposition} -- batch output không cần đếm lại
        \end{enumerate}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

% -------------------------------------------------------------------
% FRAME 3: Ba kỹ thuật
% -------------------------------------------------------------------
\begin{frame}{Ba kỹ thuật cốt lõi của LCMFreq}
  \begin{columns}[T]
    \begin{column}{0.31\textwidth}
      \begin{block}{1. BackTracking (DFS)}
        \begin{itemize}\setlength\itemsep{2pt}
          \item Xuất phát từ $P=\emptyset$, mở rộng DFS
          \item Chỉ thêm item $e > \mathrm{tail}(P)$
          \item Đảm bảo mỗi itemset xuất hiện đúng 1 lần
          \item Stack chỉ giữ đường đi hiện tại\\
            $\Rightarrow$ bộ nhớ $O(\|\mathcal{D}\|\cdot\mathrm{depth})$
        \end{itemize}
        \vspace{4pt}
        \begin{tcolorbox}[enhanced, colback=psdearblue!8, colframe=psdearblue,
          boxrule=0.4pt, arc=2pt, left=3pt, right=3pt, top=2pt, bottom=2pt]
          \fontsize{5.5}{7}\selectfont\ttfamily
          BackTracking(P):\\
          \quad output P\\
          \quad for e > tail(P):\\
          \quad\quad if freq(P$\cup$\{e\}):\\
          \quad\quad\quad BackTracking(P$\cup$\{e\})
        \end{tcolorbox}
      \end{block}
    \end{column}

    \begin{column}{0.31\textwidth}
      \begin{block}{2. OccurrenceDeliver}
        \begin{itemize}\setlength\itemsep{2pt}
          \item Một lần duyệt qua $\mathcal{T}(P)$
          \item Mỗi $t\in\mathcal{T}(P)$: phân phối vào bucket của từng item $e\in t$
          \item $\mathrm{Bucket}[e] = \mathcal{T}(P\cup\{e\})$ cho mọi $e$ cùng lúc
          \item Chi phí: $O\!\left(\sum_{t\in\mathcal{T}(P)}|t|\right)$
        \end{itemize}
        \vspace{4pt}
        \begin{tcolorbox}[enhanced, colback=psdearteal!8, colframe=psdearteal,
          boxrule=0.4pt, arc=2pt, left=3pt, right=3pt, top=2pt, bottom=2pt]
          \fontsize{5.5}{7}\selectfont\ttfamily
          for tid in T(P):\\
          \quad for e in transactions[tid]:\\
          \quad\quad Bucket[e].push(tid)
        \end{tcolorbox}
      \end{block}
    \end{column}

    \begin{column}{0.31\textwidth}
      \begin{block}{3. HypercubeDecomposition}
        \begin{itemize}\setlength\itemsep{2pt}
          \item $H(P)=\{e:|{\rm Bucket}[e]|=|{\mathcal T}(P)|\}$
          \item $\forall Q\subseteq H(P):\mathcal{T}(P\cup Q)=\mathcal{T}(P)$
          \item $S'=S\cup H(P)$; output $2^{|S'|}$ itemsets cùng support
          \item Không đệ quy cho item trong $H(P)$
        \end{itemize}
        \vspace{4pt}
        \begin{tcolorbox}[enhanced, colback=psdearorange!8, colframe=psdearorange,
          boxrule=0.4pt, arc=2pt, left=3pt, right=3pt, top=2pt, bottom=2pt]
          \fontsize{5.5}{7}\selectfont\ttfamily
          S' = S $\cup$ H(P)\\
          for Q $\subseteq$ S':\\
          \quad output P $\cup$ Q : sup(P)\\
          for e $\notin$ S', e > tail(P):\\
          \quad if freq: recurse(P$\cup$\{e\}, S')
        \end{tcolorbox}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% FRAME 7: Tối ưu hóa BitArray (Optimized version)
% -------------------------------------------------------------------
\begin{frame}{Tối ưu hóa: BitArray thay TID-list}

  \begin{columns}[T, onlytextwidth]
    \begin{column}{0.44\textwidth}
      \begin{block}{Ý tưởng}
        \begin{itemize}\setlength\itemsep{3pt}
          \item Thay \texttt{Vector\{Int\}} (danh sách TID) bằng \texttt{BitArray\{1\}} (bitmap)
          \item $\mathcal{T}(P\cup\{e\})=\mathcal{T}(P)\ \mathbf{AND}\ \mathcal{T}(\{e\})$\\
            bằng phép AND theo bit (SIMD, 64 TID/lần)
          \item $\mathrm{sup}(P)=\texttt{count(tidset\_P)}$\\
            bằng lệnh POPCNT phần cứng
        \end{itemize}
      \end{block}
      \vspace{4pt}
      \begin{exampleblock}{So sánh baseline vs optimized}
        \footnotesize\centering
        \begin{tabular}{@{}lll@{}}
          \toprule
          Phép toán & Baseline & Optimized \\
          \midrule
          Giao tidset & OccDeliver loop & \texttt{.{\&}} SIMD AND \\
          Đếm support & \texttt{length(v)} & \texttt{count(bits)} POPCNT \\
          Lưu trữ & Vector\{Int\} & BitArray\{1\} \\
          Dense data & Nhiều phần tử & Ít bit = compact \\
          \bottomrule
        \end{tabular}
      \end{exampleblock}
    \end{column}

    \begin{column}{0.52\textwidth}
      \begin{block}{Code: \texttt{\_hypercube!} (optimized)}
        \begin{tcolorbox}[enhanced, colback=psdearbg, colframe=psdearblue!40,
          boxrule=0.4pt, arc=2pt, left=4pt, right=4pt, top=2pt, bottom=2pt]
          \fontsize{6}{8}\selectfont\ttfamily
          \textcolor{psdearblue}{\# Giao tidset qua AND bit:}\\
          tidset\_Pe = tidset\_P .{\&} tidsets\_full[e]\\[3pt]
          \textcolor{psdearblue}{\# Đếm support qua POPCNT:}\\
          support\_Pe = count(tidset\_Pe)\\[3pt]
          \textcolor{psdearblue}{\# H(P) via AND:}\\
          H\_P = filter(freq\_items) do e\\
          \quad count(tidset\_P .{\&} tidsets\_full[e])\\
          \quad\quad == support\_P\\
          end\\[3pt]
          \textcolor{psdearblue}{\# Batch output (giống baseline):}\\
          for Q in \_all\_subsets(S\_prime)\\
          \quad push!(result, FreqItemset(...))\\
          end
        \end{tcolorbox}
      \end{block}
      \vspace{3pt}
      \begin{alertblock}{Kết quả thực nghiệm (tóm tắt)}
        \footnotesize
        \begin{itemize}\setlength\itemsep{2pt}
          \item Match 100\% với SPMF trên 9 tập dữ liệu chuẩn
          \item Nhanh hơn SPMF Java từ $3.5\times$ đến $141\times$
          \item Tiết kiệm bộ nhớ tới $27\times$ so với baseline
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}
